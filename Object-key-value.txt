def get_value(obj, key):
    keys = key.split('/')
    for k in keys:
        obj = obj.get(k)
        if obj is None:
            return None
    return obj

object1 = {"a":{"b":{"c":"d"}}}
key1 = 'a/b/c'
print(get_value(object1, key1)) 

object2 = {"x":{"y":{"z":"a"}}}
key2 = 'x/y/z'
print(get_value(object2, key2))
